/* --------------------------------------------------------------

   2️⃣  Render the cards (title + speakers)

   -------------------------------------------------------------- */

const container = document.getElementById('cards');

sessions.forEach((s, i) => {

  const card = document.createElement('div');

  card.className = 'card';

  card.dataset.idx = i;               // keep index for click handling

  card.innerHTML = `

    <div class="title"><strong>${s.title}</strong></div>

    <div class="speakers"><strong>${s.speakers}</strong></div>

  `;

  container.appendChild(card);

});


/* --------------------------------------------------------------

   3️⃣  Modal handling – read the pre‑filled `desc` field.

   -------------------------------------------------------------- */

const modal = document.getElementById('modal');

const modalTitle = document.getElementById('modalTitle');

const modalSpeakers = document.getElementById('modalSpeakers');

const modalDesc = document.getElementById('modalDesc');

const modalLink = document.getElementById('modalLink');

const closeBtn = document.getElementById('closeBtn');


document.addEventListener('click', e => {

  const card = e.target.closest('.card');

  if (!card) return;


  const idx = card.dataset.idx;

  const sess = sessions[idx];


  // Wrap title & speakers in <strong> for the modal as well

  modalTitle.innerHTML = `<strong>${sess.title}</strong>`;

  modalSpeakers.innerHTML = `<strong>${sess.speakers}</strong>`;

  modalDesc.innerHTML = sess.desc || '<p>No description supplied.</p>';

  modalLink.href = sess.url;

  modalLink.textContent = 'Open full session page';


  modal.style.display = 'flex';

});


closeBtn.onclick = () => modal.style.display = 'none';

modal.onclick = e => { if (e.target===modal) modal.style.display='none'; };
